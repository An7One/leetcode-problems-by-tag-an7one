#Tips
    Main Diagonal:  y = -x + b  =>   b = y + x
        isSameMainDiagonal: [row + N - col + 1] = true, N = 2 * n - 1
    Antidiagonal:   y = x + b   =>   b = y - x
        isSameAntidiagonal: [row + col] = true, N = 2 * n - 1
